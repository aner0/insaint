 function openBook(file, title) {
    document.getElementById("book-title").innerText = title;
    document.getElementById("book-window").style.display = "flex";
    document.getElementById("pdf-frame").src = file;
  }

  function closeBook() {
    document.getElementById("book-window").style.display = "none";
    document.getElementById("pdf-frame").src = ""; // очищаем
  }

  function openPdf(pdfPath, title) {
    const frame = document.getElementById('pdf-frame');
    frame.src = pdfPath;
    document.getElementById('book-title').textContent = title;
    document.getElementById('book-window').style.display = 'flex';
  }
